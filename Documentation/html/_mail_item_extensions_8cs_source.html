105  public static string GetSenderSMTPAddress(this Outlook.MailItem olItem)
106  {
107  string result = string.Empty;
108 
109  if (senderAddressCache.ContainsKey(olItem))
110  {
111  result = senderAddressCache[olItem];
112  }
113 
114  if (string.IsNullOrEmpty(result))
115  {
116  try
117  {
118  switch (olItem.SenderEmailType)
119  {
120  case "SMTP": /* an SMTP address; easy */
121  result = olItem.SenderEmailAddress;
122  break;
123  case "EX": /* an Exchange address */
124  var sender = olItem.Sender;
125  if (sender != null)
126  {
127  var exchangeUser = sender.GetExchangeUser();
128  if (exchangeUser != null)
129  {
130  result = exchangeUser.PrimarySmtpAddress;
131  }
132  }
133  break;
134  case "":
135  case null:
136  /* happens, is coped with in the final clause, don't worry about it */
137  break;
138  default:
139  Log.Warn($"Unknown email type {olItem.SenderEmailType}");
140  break;
141  }
142  }
143  catch (Exception any)
144  {
145  ErrorHandler.Handle("Failed while trying to get the sender's SMTP address", any);
146  }
147 
148  try
149  {
150  if (string.IsNullOrEmpty(result))
151  {
152  var currentUser = Globals.ThisAddIn.Application.ActiveExplorer().Session.CurrentUser.PropertyAccessor;
153  result = currentUser.GetProperty(PR_SMTP_ADDRESS).ToString();
154  }
155  }
156  catch (Exception any)
157  {
158  ErrorHandler.Handle("Failed to get email address of current user",
159  any);
160  }
161 
162  senderAddressCache[olItem] = result;
163  }
164 
165  return result;
166  }

183

198  break;
199  }
200  }
201

203

205  mailArchive.Subject = olItem.Subject;
206  mailArchive.Sent = olItem.ArchiveTime(reason);
207  mailArchive.Body = olItem.Body;
208  mailArchive.HTMLBody = Tidy(olItem.HTMLBody);
209  mailArchive.Reason = reason;

211  olItem.UserProperties[CRMCategoryPropertyName].Value :
