271  private void CacheAccessPermission(string moduleKey, string direction, bool allowed)
272  {
273  if (CRMPermissionsCache.cache.ContainsKey(moduleKey))
274  {
275  switch (CRMPermissionsCache.cache[moduleKey])
276  {
277  case SyncDirection.Direction.Neither:
278  /* shouldn't happen as it would be unwise to cache 'neither' unless
279  * we know it is true - which we won't. */
280  CacheAllowAccess(moduleKey, direction, allowed);
281  break;
282  case SyncDirection.Direction.Export:
283  if (direction == ImportPermissionToken && allowed)
284  {
285  /* if we already had export permission and now we have import permission, we have
286  * both. */
287  CRMPermissionsCache.cache[moduleKey] = SyncDirection.Direction.BiDirectional;
288  }
289  break;
290  case SyncDirection.Direction.Import:
291  if (direction == ExportPermissionToken && allowed)
292  {
293  /* if we already had import permission and now we have export permission, we have
294  * both. */
295  CRMPermissionsCache.cache[moduleKey] = SyncDirection.Direction.BiDirectional;
296  }
297  break;
298  }
299  }
300  else
301  {
302  CacheAllowAccess(moduleKey, direction, allowed);
303  }
