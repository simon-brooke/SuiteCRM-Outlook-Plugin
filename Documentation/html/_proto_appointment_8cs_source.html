106  public static CrmId TryResolveOrganiser(Outlook.AppointmentItem olItem)
107  {
108  CrmId result = CrmId.Empty;
109  string organiser = olItem.Organizer;
110 
111  try
112  {
113  if (organiser.IndexOf('@') > -1)
114  {
115  foreach (string pattern in new string[] { @".*<(.+@.+)>", @".+@.+" })
116  {
117  Match match = Regex.Match(organiser, pattern, RegexOptions.IgnoreCase);
118  if (match.Success)
119  {
120  string address = match.Groups[0].Value;
121 
122  try
123  {
124  result = CrmId.Get(RestAPIWrapper.GetUserId(new MailAddress(address)));
125  }
126  catch (FormatException)
127  {
128  // not a valid email address - no matter.
129  }
130  }
131  }
132  }
133  else
134  {
135  result = CrmId.Get(RestAPIWrapper.GetUserId(organiser));
136  }
137  }
138  catch (Exception any)
139  {
140  ErrorHandler.Handle($"Failed to resolve organiser `{olItem.Organizer}` of meeting `{olItem.Subject}`", any);
141  }
142 
143  return result;
144  }

152  public override NameValueCollection AsNameValues()
153  {
154  string statusString;
155  string name;
156 
157  switch (this.Status)

162  break;

165  name = this.subject;

174  RestAPIWrapper.SetNameValuePair("date_start",

176  RestAPIWrapper.SetNameValuePair("date_end",
177  string.Format("{0:yyyy-MM-dd HH:mm:ss}", this.end.ToUniversalTime())),
178  RestAPIWrapper.SetNameValuePair("duration_minutes", (this.duration % 60).ToString()),
179  RestAPIWrapper.SetNameValuePair("duration_hours", (this.duration / 60).ToString()),
180  RestAPIWrapper.SetNameValuePair("outlook_id", this.globalId),
181  RestAPIWrapper.SetNameValuePair("status", statusString)
182  };
183 
184  if (CrmId.IsValid(this.organiser))
185  {
186  data.Add(RestAPIWrapper.SetNameValuePair("assigned_user_id", this.organiser.ToString()));
187  }
188 
189  if (CrmId.IsInvalid(CrmEntryId))
190  {
191  /* A Guid can be constructed from a 32 digit hex string. The globalId is a
192  * 112 digit hex string. It appears from inspection that the least significant
193  * bytes are those that vary between examples, with the most significant bytes
194  * being invariant in the samples we have to hand. */
195  CrmEntryId = CrmId.Get(new Guid(this.globalId.Substring(this.globalId.Length - 32)));
196  data.Add(RestAPIWrapper.SetNameValuePair("new_with_id", true));
197  }
198 
199  data.Add(RestAPIWrapper.SetNameValuePair("id", CrmEntryId.ToString()));
200
