207  protected void PopulateCustomModulesListView(ListView view, List<string> toIgnore)
208  {
209  foreach (AvailableModule module in
210  RestAPIWrapper.GetModulesHavingEmailRelationships()
211  .OrderBy(i => i.module_key))
212  {
213  if (!toIgnore.Contains(module.module_key))
214  {
215  view.Items.Add(new ListViewItem
216  {
217  Checked = IsSelectedCustomModule(module),
218  Text = module.module_key,
219  Tag = module.module_key,
220  SubItems = { string.IsNullOrWhiteSpace(module.module_label) ?
221  module.module_key :
222  module.module_label}
223  });
224  }
225  }

233  private bool IsSelectedCustomModule(AvailableModule module)
234  {
235  return Properties.Settings.Default.CustomModules != null &&

239  private void btnSave_Click(object sender, EventArgs e)
240  {

249  Properties.Settings.Default.CustomModules.Add(item.Text + "|" + item.SubItems[1].Text);
