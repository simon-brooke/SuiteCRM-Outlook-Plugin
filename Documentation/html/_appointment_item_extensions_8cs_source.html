203  public static string GetVCalId(this Outlook.AppointmentItem item)
204  {
205  string result = string.Empty;
206 
207  //This parses the Global Appointment ID to a byte array. We need to retrieve the "UID" from it (if available).
208  byte[] bytes = (byte[])item.PropertyAccessor.StringToBinary(item.GlobalAppointmentID);
209 
210  //According to https://msdn.microsoft.com/en-us/library/ee157690(v=exchg.80).aspx we don't need first 40 bytes
211  if (bytes.Length >= 40)
212  {
213  byte[] bytesThatContainData = new byte[bytes.Length - 40];
214  Array.Copy(bytes, 40, bytesThatContainData, 0, bytesThatContainData.Length);
215 
216  //In some cases, there won't be a UID.
217  var decoded = Encoding.UTF8.GetString(bytesThatContainData, 0, bytesThatContainData.Length);
218 
219  if (decoded.StartsWith("vCal-Uid"))
220  {
221  //remove vCal-Uid from start string and special symbols
222  result = decoded.Replace("vCal-Uid", string.Empty).Replace("\u0001", string.Empty).Replace("\0", string.Empty);
223  }
224 #if DEBUG
225  else
226  {
227  // Bad format!
228  Globals.ThisAddIn.Log.Debug($"Failed to find vCal-Uid in GlobalAppointmentId '{Encoding.UTF8.GetString(bytes)}' in appointment '{item.Subject}'");
229  }
230 #endif
231  }
232  else
233  {
234  Globals.ThisAddIn.Log.Debug($"Failed to find vCal-Uid in short GlobalAppointmentId '{Encoding.UTF8.GetString(bytes)}' in appointment '{item.Subject}'");
235  }
236 
237  return result;
238  }
