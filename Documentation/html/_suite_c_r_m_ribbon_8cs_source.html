257  private static string GetResourceText(string resourceName)
258  {
259  Assembly asm = Assembly.GetExecutingAssembly();
260  string[] resourceNames = asm.GetManifestResourceNames();
261  for (int i = 0; i < resourceNames.Length; ++i)
262  {
263  if (string.Compare(resourceName, resourceNames[i], StringComparison.OrdinalIgnoreCase) == 0)
264  {
265  using (StreamReader resourceReader = new StreamReader(asm.GetManifestResourceStream(resourceNames[i])))
266  {
267  if (resourceReader != null)
268  {
269  return resourceReader.ReadToEnd();
270  }
271  }
272  }
273  }
274  return null;
275  }
