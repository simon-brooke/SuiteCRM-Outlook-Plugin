324

326  * There's a nasty little bug (#223) where Outlook offers us back in a different thread
327  * the item we're creating, before we're able to set up the sync state which marks it
328  * as already known. By locking on the enqueueing lock here, we should prevent that.
329  */
330  lock (enqueueingLock)
331  {
332  try
333  {

339  olItem.Body = crmItem.GetValueAsString("description");
340  SetMeetingStatus(olItem, crmItem);
341 
342  /* set the SEntryID property quickly, create the sync state and save the item, to reduce howlaround */
343  EnsureSynchronisationPropertiesForOutlookItem(olItem, crmItem, crmType);
344 
345  LogItemAction(olItem, "AppointmentSyncing.AddNewItemFromCrmToOutlook");
346  if (!string.IsNullOrWhiteSpace(crmItem.GetValueAsString("date_start")))
347  {
348  olItem.Start = dateStart;
349  SetOutlookItemDuration(crmType, crmItem, olItem);
350  }
351  }
352  catch (Exception any)
353  {
354  Log.Warn("Unexpected error in AppointmentSyncing.AddNewItemFromCrmToOutlook", any);
355  }
356  finally
357  {
358  if (olItem != null)
359  {
360  try
361  {
362  olItem.Save();
363  newState = SyncStateManager.Instance.GetOrCreateSyncState(olItem) as SyncStateType;
364  newState.SetNewFromCRM();
365  }
366  catch (COMException comx)
367  {
368  // this really, really shouldn't happen!
369  ErrorHandler.Handle(comx);

626  protected override SyncState<AppointmentItem> AddOrUpdateItemFromCrmToOutlook(
627  MAPIFolder folder,
628  string crmType,
629  EntryValue crmItem)
630  {
631  var existing = SyncStateManager.Instance.GetExistingSyncState(crmItem);
632  var result = existing as SyncStateType;
633 
634  var dateStart = crmItem.GetValueAsDateTime("date_start");
635 
636  if (dateStart >= GetStartDate())
637  {
638  /* search for the item among the sync states I already know about */
639  if (existing == null)
640  {
641  /* check for howlaround */
642  var matches = FindMatches(crmItem);
643 
644  if (!matches.Any())
645  {
646  /* didn't find it, so add it to Outlook */
647  result = AddNewItemFromCrmToOutlook(folder, crmType, crmItem, dateStart);
648  }
649  else
650  {
651  var withoutCrmId = matches.Where(x => CrmId.IsInvalid(x.CrmEntryId)).ToList();
652  var crmId = CrmId.Get(crmItem.id);
653  if (withoutCrmId.Any())
654  {
655  result = withoutCrmId.ElementAt(0) as SyncStateType;
656  if (result != null)
657  {
658  result.CrmEntryId = crmId;
659  result.OutlookItem.SetCrmId(crmId);
660  UpdateExistingOutlookItemFromCrm(crmType, crmItem, dateStart, result);
661  }
662  }
663  else
664  {
665  result = matches.ElementAt(0) as SyncStateType;
666  Log.Warn(
667  $"Howlaround detected? Appointment '{crmItem.GetValueAsString("name")}' offered with id {crmId}, expected {matches[0].CrmEntryId}, {matches.Count} duplicates");
668  }
669  }
670  }
671  else if (result != null)
672  {
673  /* found it, so update it from the CRM item */
674  UpdateExistingOutlookItemFromCrm(crmType, crmItem, dateStart, result);
675  }
676  else
677  {
678  throw new UnexpectedSyncStateClassException($"{GetType().Name}", existing);
679  }
680 
681  existing?.SaveItem();
682  }
683 
684  return result;
