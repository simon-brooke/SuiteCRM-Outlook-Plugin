335  private SyncState<Outlook.TaskItem> AddNewItemFromCrmToOutlook(Outlook.MAPIFolder tasksFolder, EntryValue crmItem)
336  {
337  TaskSyncState result = null;
338  Log.Debug(
339  (string)string.Format(
340  $"{this.GetType().Name}.AddNewItemFromCrmToOutlook, entry id is '{crmItem.GetValueAsString("id")}', creating in Outlook."));
341 
342  /*
343  * There's a nasty little bug (#223) where Outlook offers us back in a different thread
344  * the item we're creating, before we're able to set up the sync state which marks it
345  * as already known. By locking on the enqueueing lock here, we should prevent that.
346  */
347  lock (enqueueingLock)
348  {
349  Outlook.TaskItem olItem = tasksFolder.Items.Add(Outlook.OlItemType.olTaskItem);
350

354  {
355  this.SetOutlookItemPropertiesFromCrmItem(crmItem, olItem);
356  }
